Get it from Google Place ID Finder. const GOOGLE_PLACE_ID = 'PLACE_ID_HERE'; function renderStars(container, rating){ const full = Math.floor(rating||0), half = (rating - full) >= 0.5; container.innerHTML = ''; for(let i=0;i<5;i++){ const s = document.createElement('span'); s.className='star'; s.style.opacity = i < full ? 1 : (i===full && half ? .5 : .2); container.appendChild(s); } } function initGoogleReviews(){ if(!GOOGLE_PLACE_ID || GOOGLE_PLACE_ID==='PLACE_ID_HERE') return; // keep fallback visible const service = new google.maps.places.PlacesService(document.createElement('div')); service.getDetails({placeId: GOOGLE_PLACE_ID, fields: ['name','rating','user_ratings_total','reviews']}, (place, status)=>{ if(status !== google.maps.places.PlacesServiceStatus.OK || !place) return; const avg = place.rating || 0; const total = place.user_ratings_total || 0; document.getElementById('avg-rating').textContent = avg.toFixed(1); document.getElementById('ratings-total').textContent = total + ' Google reviews'; renderStars(document.getElementById('avg-stars'), avg); const list = document.getElementById('reviews-list'); list.innerHTML = ''; const reviews = (place.reviews||[]).slice().sort((a,b)=> (b.time||0)-(a.time||0)).slice(0,6); reviews.forEach(r=>{ const card = document.createElement('div'); card.className='reviews-card hover-rise'; const head = document.createElement('div'); head.style.display='flex'; head.style.justifyContent='space-between'; head.style.alignItems='center'; const who = document.createElement('strong'); who.textContent = r.author_name || 'Anonymous'; const stars = document.createElement('div'); renderStars(stars, r.rating||0); head.appendChild(who); head.appendChild(stars); card.appendChild(head); const text = document.createElement('p'); text.className='muted'; text.style.marginTop='.6rem'; const full = r.text || ''; const short = full.length>260 ? full.slice(0,260)+'…' : full; text.textContent = short; card.appendChild(text); if(full.length>260){ const btn = document.createElement('button'); btn.textContent='Read more'; btn.className='cta-btn'; btn.style.padding='.35rem .6rem'; btn.style.fontWeight='600'; btn.style.fontSize='.9rem'; btn.addEventListener('click',()=>{ text.textContent = full; btn.remove(); }); card.appendChild(btn); } list.appendChild(card); }); // hide fallback map if we have reviews if(reviews.length){ const fb = document.getElementById('reviews-fallback'); if(fb) fb.style.display='none'; } }); }
